  • еҺҹзҗҶпјҡ

 з®ҖеҚ•зүҲпјҡжҢҮеҶ…ж ёеңЁеҶ…ж ёжҖҒдёӢеҫӘзҺҜи¶…иҝҮXXз§’пјҢдёҚз»ҷе…¶д»–иҝӣзЁӢиҝҗиЎҢзҡ„жңәдјҡгҖӮе…¶е®һе°ұжҳҜеҶ…ж ёдјҡе®ҡж—¶жЈҖжөӢеҶ…ж ёжҳҜеҗҰиҝҳиғҪи°ғеәҰпјҢеҰӮжһңи¶…иҝҮдёҖж®өж—¶й—ҙдёҚиғҪи°ғеәҰпјҢйӮЈе°ұи®ӨдёәsoftlockupдәҶгҖӮ

 зЁҚеҫ®еӨҚжқӮзүҲпјҡеҰӮжһңй…ҚзҪ®дәҶCONFIG_SOFTLOCKUP_DETECTORпјҢйӮЈд№ҲеҶ…ж ёеңЁеҗҜеҠЁж—¶е°ұдјҡиө·дёҖдёӘhrtimerе®ҡж—¶еҷЁгҖӮиҝҷдёӘе®ҡж—¶еҷЁжҜҸйҡ”watchdog_threshпјҲе°ұжҳҜдёҠйқўзҡ„еҶ…ж ёеҸӮж•°пјүе°ұе”ӨйҶ’дёҖж¬ЎпјҢдҪңз”ЁжңүдёӨдёӘпјҡдёҖпјҢе”ӨйҶ’percpuзҡ„д»»еҠЎеҺ»жӣҙж–°ж—¶й—ҙжҲіпјӣдәҢпјҢжЈҖжөӢжҳҜеҗҰеҸ‘з”ҹдәҶlockupгҖӮжүҖд»ҘпјҢеҰӮжһңжңүдёҖз§ҚеңәжҷҜпјҢжҜ”еҰӮпјҢдёҖдёӘзәҝзЁӢиҝӣе…ҘеҶ…ж ёжҖҒиҮӘж—Ӣзӯүй”ҒпјҢдёҚйҮҠж”ҫcpuпјҢhrtimerе®ҡж—¶еҷЁе°қиҜ•еҗ‘жҜҸдёӘcpuеҸ‘stop_cpuд»»еҠЎеҺ»жӣҙж–°ж—¶й—ҙжҲігҖӮдҪҶжҳҜеӣ дёәеҶ…ж ёдёҖзӣҙеңЁеҶ…ж ёжҖҒиҮӘж—ӢжІЎеҠһжі•и°ғеәҰеҲ°жӣҙж–°ж—¶й—ҙжҲізҡ„д»»еҠЎеҺ»пјҢдәҺжҳҜе°ұдёҚиғҪжӣҙж–°ж—¶й—ҙжҲіпјҢеҪ“и¶…иҝҮwatchdog_thresh*2ж—¶й—ҙиҝҳжІЎжңүжӣҙж–°ж—¶й—ҙжҲіж—¶пјҢhrtimerе®ҡж—¶еҷЁиў«е”ӨйҶ’е°ұдјҡеҲӨж–ӯиҝҷж—¶еҖҷеҸ‘з”ҹдәҶsoftlockupпјҢеҰӮжһңиҝҷж—¶еҖҷиҝҳй…ҚзҪ®дәҶsoftlockup_panicпјҢзі»з»ҹе°ұдјҡи§ҰеҸ‘panicдәҶгҖӮ

void __init lockup_detector_init(void)
{
    //еҰӮжһңжңүејҖеҗҜnohz_fullзҡ„cpuе°ұи·іиҝҮ
    if (tick_nohz_full_enabled())
        pr_info("Disabling watchdog on nohz_full cores by default\n");
    //з„¶еҗҺиҺ·еҸ–йқһisolatedзҡ„cpuпјҢжіЁж„ҸиҝҷйҮҢзҡ„housekeeping_cpumaskиҝ”еӣһзҡ„жҳҜйқһisolatedзҡ„cpu mask
    //дҪҶжҳҜиҝҷдёӘtypeдёҚжё…жҘҡд»Җд№Ҳж„ҸжҖқпјҢйңҖиҰҒзңӢзңӢ
    cpumask_copy(&watchdog_cpumask,
                 housekeeping_cpumask(HK_TYPE_TIMER));
    //еҲӨж–ӯnmiжҳҜеҗҰеҸҜз”Ё
    //дёҖиҲ¬еҸҜз”Ёдё”дёҖиҲ¬жҳҜз”Ёnmiзҡ„
    if (!watchdog_nmi_probe())
        nmi_watchdog_available = true;
    lockup_detector_setup();
    //иҝҷжҳҜеҺ»еҲқе§ӢеҢ–sysctlеҸӮж•°пјҢд№ҹе°ұжҳҜдёҠйқўжҸҗеҲ°зҡ„зӣёе…ізҡ„еҶ…ж ёеҸӮж•°
    watchdog_sysctl_init();
}

//е…ҲеҸӘзңӢsoftlockup
softlockup_start_all

static void softlockup_start_all(void)
{
    int cpu;

    cpumask_copy(&watchdog_allowed_mask, &watchdog_cpumask);
    //з»ҷжҜҸдёӘcpuеҸ‘дёҖдёӘsoftlockup_start_fnзҡ„ipi
    for_each_cpu(cpu, &watchdog_allowed_mask)
        smp_call_on_cpu(cpu, softlockup_start_fn, NULL, false);
}

static int softlockup_start_fn(void *data)
{
    watchdog_enable(smp_processor_id());
    return 0;
}

//watchdog_enableйҮҢдё»иҰҒе°ұжҳҜеҲқе§ӢеҢ–е’Ңи®ҫзҪ®hrtimerпјҢжү§иЎҢзҡ„жҳҜwatchdog_timer_fnеҮҪж•°
//и®ҫзҪ®е®Ңhrtimerе№¶еҗҜеҠЁд№ӢеҗҺпјҢжҜҸдёӘcpuдёҠе°ұдјҡе®ҡж—¶дә§з”ҹhrtimerдёӯж–ӯз„¶еҗҺеҺ»жү§иЎҢwatchdog_timer_fnеҮҪж•°дәҶ
//иҝҷйҮҢиҝҳдјҡжү§иЎҢеҲ°watchdog_hardlockup_enableпјҹжҢүзҗҶжқҘиҜҙsoftlockupе’ҢhardlockupдёҚеә”иҜҘиҖҰеҗҲеҗ§пјҹ
static void watchdog_enable(unsigned int cpu)
{
	struct hrtimer *hrtimer = this_cpu_ptr(&watchdog_hrtimer);
	struct completion *done = this_cpu_ptr(&softlockup_completion);

	WARN_ON_ONCE(cpu != smp_processor_id());

	init_completion(done);
	complete(done);

	hrtimer_init(hrtimer, CLOCK_MONOTONIC, HRTIMER_MODE_REL_HARD);
	hrtimer->function = watchdog_timer_fn;
	hrtimer_start(hrtimer, ns_to_ktime(sample_period),
		      HRTIMER_MODE_REL_PINNED_HARD);

	/* Initialize timestamp */
	update_touch_ts();
	/* Enable the perf event */
	if (watchdog_enabled & NMI_WATCHDOG_ENABLED)
		watchdog_nmi_enable(cpu);
}

//иҝҷйҮҢиҙҹиҙЈи®ҫзҪ®ж—¶й—ҙжҲіе№¶жЈҖжҹҘпјҢеҰӮжһңжҳҜsoftlockupдәҶпјҢе°ұи§ҰеҸ‘panicдәҶ
//зңӢдёҖдёӢдё»иҰҒжөҒзЁӢ
static enum hrtimer_restart watchdog_timer_fn(struct hrtimer *hrtimer)
{
    //еҰӮжһңwatchdogиў«е…ій—ӯдәҶпјҢе°ұдёҚеҶҚжү§иЎҢе®ҡж—¶еҷЁеҮҪж•°дәҶ
	if (!watchdog_enabled)
		return HRTIMER_NORESTART;

	/* kick the hardlockup detector */
    //hardlockupзӣёе…ізҡ„пјҢзңӢhardlockupзҡ„ж—¶еҖҷеҶҚиҜҙ
	watchdog_interrupt_count();

	/* kick the softlockup detector */
    //иҝҷеқ—е°ұжҳҜеҺ»иҰҒжү§иЎҢдёҖдёӘstop_cpuд»»еҠЎпјҢиҝҷдёӘд»»еҠЎе°ұжҳҜsoftlockup_fn
    //softlockup_fnзҡ„дё»иҰҒдҪңз”Ёе°ұжҳҜжӣҙж–°ж—¶й—ҙжҲі
	if (completion_done(this_cpu_ptr(&softlockup_completion))) {
		reinit_completion(this_cpu_ptr(&softlockup_completion));
		stop_one_cpu_nowait(smp_processor_id(),
				softlockup_fn, NULL,
				this_cpu_ptr(&softlockup_stop_work));
	}

	/* .. and repeat */
    //hrtimerзӣёе…іпјҢжҺЁиҝӣеҲ°жңҹж—¶й—ҙ
	hrtimer_forward_now(hrtimer, ns_to_ktime(sample_period));

    //дёҖдәӣиҺ·еҸ–ж—¶й—ҙжҲіж“ҚдҪң

    //еҲӨж–ӯжҳҜеҗҰsoftlockupдәҶ
    //е…¶е®һе°ұжҳҜеҲӨж–ӯж—¶й—ҙе·®жҳҜеҗҰи¶…иҝҮдәҶwatchdog_thresh * 2
	duration = is_softlockup(touch_ts, period_ts, now);
	if (unlikely(duration)) {
        //еҰӮжһңеҸ‘з”ҹдәҶsoftlockupдәҶ
        //е°ұиҝӣиЎҢеҗҺз»ӯеӨ„зҗҶпјҢжҜ”еҰӮdump stackд№Ӣзұ»
		pr_emerg("BUG: soft lockup - CPU#%d stuck for %us! [%s:%d]\n",
			smp_processor_id(), duration,
			current->comm, task_pid_nr(current));
		print_modules();
		print_irqtrace_events(current);
		if (regs)
			show_regs(regs);
		else
			dump_stack();

		add_taint(TAINT_SOFTLOCKUP, LOCKDEP_STILL_OK);
		if (softlockup_panic)
            //еҰӮжһңи®ҫзҪ®дәҶsoftlockup_panicпјҢеҲҷи§ҰеҸ‘panic
			panic("softlockup: hung tasks");
	}

	return HRTIMER_RESTART;
}

д№ӢеүҚзңӢдёҖдәӣиө„ж–ҷзҡ„ж—¶еҖҷпјҢиҜҙзҡ„жҳҜйҖҡиҝҮhrtimerйҖҡиҝҮе”ӨйҶ’еҶ…ж ёзәҝзЁӢжқҘжӣҙж–°ж—¶й—ҙжҲіпјҢе°ұжҳҜйӮЈдәӣ[watchdog/0]д№Ӣзұ»зҡ„еҶ…ж ёзәҝзЁӢпјҢжҖҺд№ҲеҲҶжһҗе®ҢжІЎжңүзңӢеҲ°еҶ…ж ёзәҝзЁӢзҡ„дәӢе‘ўпјҹ

еҺҹжқҘжҳҜзҺ°еңЁдёҚз”ЁеҶ…ж ёзәҝзЁӢдәҶпјҢд№ҹе°ұжҳҜж–°зҡ„еҶ…ж ёйҮҢе·Із»ҸзңӢдёҚеҲ°watchdog/XXXдәҶпјҢеңЁ18е№ҙзҡ„дёҖдёӘpacthйҮҢе·Із»Ҹиў«cpu_stop_workд»ЈжӣҝдәҶ

pantchпјҡ watchdog/softlockup: Replace вҖңwatchdog/%uвҖқ threads with cpu_stop_work

иҝҷж ·еӯҗжңүд»Җд№ҲеҘҪеӨ„е‘ўпјҹеҮҸе°‘дәҶper cpuзәҝзЁӢпјҹйҒҝе…ҚдәҶsoftlockupзҡ„жЈҖжөӢе’ҢSCHED_DEADLINEзҡ„иҖҰеҗҲпјҹпјҹ

hardlockup

{
  //еҰӮжһңејҖеҗҜдәҶwatchdogдё”ејҖеҗҜдәҶNMI_WATCHDOG_ENABLEDе°ұи°ғз”Ё

	if (watchdog_enabled & NMI_WATCHDOG_ENABLED)
		watchdog_nmi_enable(cpu);
}

static int hardlockup_detector_event_create(void)
{

	WARN_ON(!is_percpu_thread());
	cpu = raw_smp_processor_id();
	wd_attr = &wd_hw_attr;
  //иҺ·еҸ–sample_period
	wd_attr->sample_period = hw_nmi_get_sample_period(watchdog_thresh);
  //жіЁеҶҢдёҖдёӘзЎ¬д»¶зҡ„дәӢд»¶performance monitoringпјҢиҝҷдёӘзЎ¬д»¶жңүдёӘеҠҹиғҪе°ұжҳҜеңЁcpu clockз»ҸиҝҮеӨҡе°‘е‘Ёжңҹд№ӢеҗҺеҸ‘дёҖдёӘnmiдёӯж–ӯеҮәжқҘ
  //дёӯж–ӯзҡ„handlerе°ұжҳҜwatchdog_overflow_callbackеҮҪж•°
	/* Try to register using hardware perf events */
	evt = perf_event_create_kernel_counter(wd_attr, cpu, NULL,
					       watchdog_overflow_callback, NULL);
}


static void watchdog_overflow_callback(struct perf_event *event,
				       struct perf_sample_data *data,
				       struct pt_regs *regs)
{
//иҝҷйҮҢе°ұз”Ёis_hardlockupжЈҖжҹҘеҰӮжһңжҳҜhardlockupдәҶе°ұдә§з”ҹnmi_panic
//жЈҖжҹҘзҡ„жҳҜд»Җд№Ҳе‘ўпјҹе°ұжҳҜpercpuзҡ„hrtimerзҡ„дёӯж–ӯж•°жҳҜеҗҰеңЁеўһй•ҝ
//д»Җд№Ҳж—¶еҖҷеўһй•ҝе‘ўпјҹе”үпјҹе°ұжҳҜд№ӢеүҚwatchdog_timer_fnдёӯи°ғз”Ёзҡ„watchdog_interrupt_count
//жүҖд»ҘиҝҷйҮҢе·®дёҚеӨҡеҸҜд»ҘзЎ®е®ҡпјҢhardlockupзҡ„жЈҖжөӢдҫқиө–softlockupгҖӮдёәд»Җд№Ҳе‘ўпјҹеӣ дёәhardlockupжҳҜжЈҖжөӢзі»з»ҹиғҪдёҚиғҪе“Қеә”дёӯж–ӯ
//йӮЈеҲӨж–ӯд»Җд№Ҳдёӯж–ӯе‘ўпјҹиҮӘз„¶е°ұжҳҜsoftlockupзҡ„hrtimerдә§з”ҹзҡ„дёӯж–ӯжңҖеҰҷ
	if (is_hardlockup()) {
		int this_cpu = smp_processor_id();
    
    //еҗҢж ·пјҢеҲӨж–ӯдёәhardlockupдәҶе°ұеҒҡеҗҺз»ӯе·ҘдҪңпјҢжү“еҚ°ж Ҳзӯү
		pr_emerg("Watchdog detected hard LOCKUP on cpu %d\n",
			 this_cpu);
		print_modules();
		print_irqtrace_events(current);
		if (regs)
			show_regs(regs);
		else
			dump_stack();

    //и§ҰеҸ‘nmiдёӯж–ӯ
		if (hardlockup_panic)
			nmi_panic(regs, "Hard LOCKUP");

		__this_cpu_write(hard_watchdog_warn, true);
		return;
	}

	__this_cpu_write(hard_watchdog_warn, false);
	return;
}
